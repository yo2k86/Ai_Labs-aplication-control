function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = sheet.getDataRange().getValues();
  var action = e.parameter.action;

  // Fungsi kebal bahasa untuk membaca centangan (TRUE/BENAR)
  function isTrue(val) {
    if (val === true) return true;
    if (typeof val === 'string' && (val.toUpperCase() === "TRUE" || val.toUpperCase() === "BENAR")) return true;
    return false;
  }

  if (action === "readAll") {
    var result = [];
    for (var i = 1; i < data.length; i++) {
      result.push({
        row: i + 1,
        email: data[i][0] || "",
        username: data[i][1] || "",
        beriAkses: isTrue(data[i][2]),
        matikanAkses: isTrue(data[i][3]),
        hapusAkun: isTrue(data[i][4]),
        firstLogin: data[i][5] || ""
      });
    }
    return ContentService.createTextOutput(JSON.stringify({success: true, data: result})).setMimeType(ContentService.MimeType.JSON);
  }

  var emailParam = e.parameter.email;
  if(emailParam) {
    var response = { success: false, message: "Email tidak terdaftar di Database." };
    for (var i = 1; i < data.length; i++) { 
      var email = data[i][0];       
      if (email.toString().toLowerCase() === emailParam.toString().toLowerCase()) {
        var username = data[i][1];     
        var beriAkses = isTrue(data[i][2]);    
        var matikanAkses = isTrue(data[i][3]); 
        var hapusAkun = isTrue(data[i][4]);    
        var firstLogin = data[i][5];   

        if (hapusAkun) {
          response = { success: false, message: "Akun Anda telah dihapus permanen." };
        } else if (matikanAkses) {
          response = { success: false, message: "Akses Anda dinonaktifkan oleh sistem." };
        } else if (beriAkses) {
          if (!firstLogin || firstLogin === "") {
            var now = new Date();
            var formattedDate = Utilities.formatDate(now, "Asia/Jakarta", "dd/MM/yyyy HH:mm:ss");
            sheet.getRange(i + 1, 6).setValue(formattedDate); 
          }
          response = { success: true, email: email, username: username };
        } else {
          response = { success: false, message: "Menunggu persetujuan Admin." };
        }
        break; 
      }
    }
    return ContentService.createTextOutput(JSON.stringify(response)).setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput(JSON.stringify({success:false, message:"Parameter tidak valid"})).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  try {
    var requestBody = JSON.parse(e.postData.contents);
    var action = requestBody.action;

    // FUNGSI TAMBAH USER BARU
    if (action === "add") {
      var data = requestBody.data;
      sheet.appendRow([data.email, data.username, false, false, false, ""]);
      return ContentService.createTextOutput(JSON.stringify({success: true, message: "User ditambahkan"})).setMimeType(ContentService.MimeType.JSON);
    }
    
    // FUNGSI UPDATE DATA LAMA
    if (action === "update") {
      var updates = requestBody.updates; 
      for(var i=0; i < updates.length; i++) {
         var row = updates[i].row;
         sheet.getRange(row, 1).setValue(updates[i].email);        // Kolom A
         sheet.getRange(row, 2).setValue(updates[i].username);     // Kolom B
         sheet.getRange(row, 3).setValue(updates[i].beriAkses);    // Kolom C
         sheet.getRange(row, 4).setValue(updates[i].matikanAkses); // Kolom D
         sheet.getRange(row, 5).setValue(updates[i].hapusAkun);    // Kolom E
      }
      return ContentService.createTextOutput(JSON.stringify({success: true, message: "Berhasil update"})).setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({success: false, message: err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}
